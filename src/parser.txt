funcion PROMEDIO(entero: numeroDeElementos) 
	variables: entero: i; flotante: promedio, elemento
	
	i = 0
	
	mientras (i < numeroDeElementos)
		leer elemento
		promedio = promedio + elemento
		i = i + 1
	fin-mientras
	
	promedio = promedio / numeroDeElementos
	regresa promedio
fin-funcion
